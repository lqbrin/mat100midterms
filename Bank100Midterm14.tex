\newproblem{14a}
{
	Solve for $x$.\par
	$\displaystyle \frac{3x}{x^2-1}=\frac{7}{x+1} - \frac{5}{x-1}$
}
{
	\begin{tabular}{l r}
	$(x+1)(x-1)\frac{3x}{(x+1)(x-1)}=\frac{(x+1)(x-1)7}{x+1} - \frac{(x+1)(x-1)5}{x-1}$ & 2 pts to here\\
	$3x=7(x-1)-5(x+1)$ & 3 pts to here\\
	$3x=7x-7-5x-5$ & 4 pts to here\\
	$3x=2x-12$ & 5 pts to here\\
	$x=-12$ & 6 pts to here
	\end{tabular}
}
