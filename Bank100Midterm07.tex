\newproblem{07a}
{
	Simplify.\par
	$\displaystyle \frac{x^2-9x+18}{2x^2-9x+9}$
}
{
	\begin{tabular}{l r}

Partial factoring of only the denominator & 2 pts to here\\

$\frac{x-6}{2x-3}$ & 4 pts to here
\end{tabular}
}
